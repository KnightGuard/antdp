@antdp/basic-layouts
---

入口公共界面

## Installation

```bash
npm i @antdp/basic-layouts --save
```

## Basic Usage

```jsx
import BasicLayout from '@antdp/basic-layouts';
import logo from './logo.svg';

export default (props) => {
  return (
    <BasicLayout
      {...props}
      projectName="Ant Design Pro2"
      logo={logo}
    />
  )
};
```
